00043 class FIRMValidityChecker : public ompl::base::StateValidityChecker
00044 {
00045   public:
00046     typedef ObservationModelMethod::ObservationModelPointer ObservationModelPointer;
00047     typedef SE2BeliefSpace::StateType StateType;

00049     FIRMValidityChecker(const ompl::base::SpaceInformationPtr &si) :
00050     ompl::base::StateValidityChecker(si)
00051     {
00052     }
00053 
00054     virtual bool isValid(const ompl::base::State *state) const
00055     {
00056       // states within a box are invalid
00057       int x_l =  5;
00058       int x_r =  15;
00059       int y_b =  5;
00060       int y_t =  15;
00061 
00062       arma::colvec pos = state->as<StateType>()->getArmaData();
00063 
00064       if(pos[0] >= x_l && pos[0] <= x_r )
00065       {
00066         if(pos[1] >= y_b && pos[1] <= y_t)
00067         {
00068             return false;
00069         }
00070       }
00071       return true;
00072     }
